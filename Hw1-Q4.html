ELE 124 Computer Programming Hw1-Q4

Q4- Now solve the same problem using functions. Use the following prototypes:

a) int computeHomeValue (int popularity, int size)
b) void computeHomeValue (void)

Notice that for part b, all printf/scanf should take place within the function.

Answer a:

https://github.com/mert-ozhan/HW1-Q4.a 

                                
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int computeHomeValue(popularity, size);

int main() {
	
	int popularity, size;
	
	printf("Enter popularity: ");
	scanf("%d", &popularity);
	
	printf("Enter size: ");
	scanf("%d", &size);
	
	printf("Home value is: %d", computeHomeValue(popularity, size)); 	
	
	return 0;

}

int computeHomeValue(int popularity, int size) {

	return ((pow(popularity, 3) + pow(size, 2)) * 10000);

}

Answer b:

https://github.com/mert-ozhan/HW1-Q4.b 

                                
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

void computeHomeValue(void);

int main() {
	
	computeHomeValue();
	return 0;

}

void computeHomeValue(void) {
	
	int popularity, size, homeValue;
	
	printf("Enter popularity: ");
	scanf("%d", &popularity);
	
	printf("Enter size: ");
	scanf("%d", &size);
	
	homeValue = (pow(popularity, 3) + pow(size, 2)) * 10000;
	
	printf("Home value is: %d", homeValue);

}
